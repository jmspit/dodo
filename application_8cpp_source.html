73  Application::HostType Application::detectHostType() {
74  const std::string initcgroup = "/proc/1/cgroup";
75  if ( common::fileReadAccess( initcgroup ) ) {
76  std::vector res = fileReadStrings( initcgroup, std::regex("^0::/docker$") );
77  if ( res.size() == 1 ) return HostType::Docker;
78  }
79 
80  const std::string product_name = "/sys/class/dmi/id/product_name";
81  if ( common::fileReadAccess( product_name ) ) {
82  std::vector res = fileReadStrings( product_name, std::regex("^VirtualBox$") );
83  if ( res.size() == 1 ) return HostType::VirtualBox;
84  }
85 
86  if ( common::fileReadAccess( product_name ) ) {
87  std::vector res = fileReadStrings( product_name, std::regex("^VMware Virtual Platform$") );
88  if ( res.size() == 1 ) return HostType::VMWare;
89  }
90 
91  if ( common::fileReadAccess( product_name ) ) {
92  std::vector res = fileReadStrings( product_name, std::regex("^KVM$") );
93  if ( res.size() == 1 ) return HostType::KVM;
94  }
95 
96  const std::string cpuinfo = "/proc/cpuinfo";
97  if ( common::fileReadAccess( cpuinfo ) ) {
98  std::vector res = fileReadStrings( cpuinfo, std::regex("^flags.*:.*hypervisor") );
99  if ( res.size() > 0 ) return HostType::GenericVM;
100  }
101 
102  return HostType::BareMetal;
103  }
