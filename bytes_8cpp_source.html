33  void Bytes::reserve( size_t sz ) {
34  size_t chunkedsz = ( sz / alloc_block + 1 ) * alloc_block;
35  if ( sz == size_ ) return;
36  if ( array_ ) {
37  if ( sz > allocated_size ) {
38  array_ = static_cast< Octet*>( std::realloc( array_, chunkedsz ) );
39  if ( array_ || sz == 0 ) {
40  size_ = sz;
41  allocated_size = chunkedsz;
42  if ( !sz ) array_ = nullptr;
43  } else throw_SystemException( "realloc of " << sz << " bytes failed", errno );
44  } else {
45  size_ = sz;
46  }
47  } else {
48  array_ = static_cast< Octet*>( std::malloc( chunkedsz ) );
49  if ( array_ ) {
50  size_ = sz;
51  allocated_size = chunkedsz;
52  } else throw_SystemException( "malloc of " << sz << " bytes failed", errno );
53  }
54  }

114  Bytes& Bytes::decodeBase64( const std::string& src ) {
115  Octet* temp = (Octet*)std::malloc( src.length() ); // too much, but safe, and temp anyway
116 
117 
118  EVP_ENCODE_CTX* ctx = EVP_ENCODE_CTX_new();
119  if ( !ctx ) throw_Exception( "EVP_ENCODE_CTX_new failed" );
120  EVP_DecodeInit( ctx );
121  int len = 0;
122  int rc = EVP_DecodeUpdate( ctx, temp, &len, (const unsigned char*)src.c_str(), (int)src.length() );
123  if ( rc == -1 ) throw_Exception( "EVP_DecodeUpdate failed" );
124  int actsize = len;
125  rc = EVP_DecodeFinal( ctx, temp + len, &len );
126  if ( rc != 1 ) throw_Exception( "EVP_DecodeFinal failed" );
127  actsize += len;
128  EVP_ENCODE_CTX_free( ctx );
129 
130  this->reserve( actsize );
131  memcpy( array_, temp, actsize );
132  std::free( temp );
133  return *this;
134  }

162  Bytes::MatchType Bytes::match( const Bytes& other, size_t index, size_t &octets ) {
163  size_t local_size = size_ - index;
164  size_t match_size = std::min( local_size, other.size_ );
165  int cmp = memcmp( array_, other.array_, match_size );
166  if ( cmp == 0 ) {
167  if ( local_size < other.size_ ) {
168  octets = local_size;
169  return MatchType::Contained;
170  } else if ( local_size > other.size_ ) {
171  octets = other.size_;
172  return MatchType::Contains;
173  } else {
174  octets = local_size;
175  return MatchType::Full;
176  }
177  } else {
178  octets = 0;
179  return MatchType::Mismatch;
180  }
181  }
