99  HTTPMessage::ParseResult HTTPRequest::parse( VirtualReadBuffer &data ) {
100  ParseResult parseResult = request_line_.parse( data );
101  if ( parseResult.ok() ) {
102  if ( data.get() == charCR ) {
103  parseResult.setSystemError( data.next() );
104  if ( parseResult.ok() && data.get() == charLF ) {
105  parseResult.setSystemError( data.next() );
106  if ( parseResult.ok() ) return parseBody( data ); else return parseResult;
107  } else return { peOk, common::SystemError::ecOK };
108  } else {
109  // these are headers
110  parseResult = parseHeaders( data );
111  if ( !parseResult.ok() ) return parseResult;
112  if ( data.get() == charLF ) {
113  parseResult.setSystemError( data.next() );
114  if ( parseResult.ok() ) {
115  return parseBody( data );
116  } else if ( parseResult.eof() ) return { peOk, common::SystemError::ecOK };
117  else return parseResult;
118  } else return { peOk, common::SystemError::ecOK };
119  }
120  }
121  return parseResult;
122  }

124  HTTPMessage::ParseResult HTTPRequest::HTTPRequestLine::parse( VirtualReadBuffer &data ) {
125  enum ParseState {
126  psMethodStart,
127  psMethodEnd,
128  psRequestURIStart,
129  psRequestURIEnd,
130  psHTTPVersionCR,
131  psHTTPVersionLF,
132  psHTTPVersionEnd,
133  psError,
134  psDone,
135  };
136  ParseState state = psMethodStart;
137  std::string method_string = "";
138  std::string version_string = "";
139  request_uri_ = "";
140  ParseResult parseResult;
141  while ( state != psDone && state != psError && parseResult.ok() ) {
142  switch ( state ) {
143 
144  case psMethodStart:
145  if ( std::isspace( data.get() ) ) {
146  state = psMethodEnd;
147  } else {
148  method_string += data.get();
149  parseResult.setSystemError( data.next() );
150  if ( ! parseResult.ok() ) return parseResult;
151  };
152  break;
153 
154  case psMethodEnd:
155  parseResult.setSystemError( eatSpace( data ) );
156  if ( ! parseResult.ok() ) return parseResult;
157  state = psRequestURIStart;
158  break;
159 
160  case psRequestURIStart:
161  if ( std::isspace( data.get() ) ) {
162  state = psRequestURIEnd;
163  } else {
164  request_uri_ += data.get();
165  parseResult.setSystemError( data.next() );
166  if ( ! parseResult.ok() ) return parseResult;
167  }
168  break;
169 
170  case psRequestURIEnd:
171  parseResult = http_version_.parse( data );
172  if ( parseResult.ok() ) {
173  state = psHTTPVersionCR;
174  parseResult.setSystemError( eatSpace( data ) );
175  } else return parseResult;
176  break;
177 
178  case psHTTPVersionCR:
179  if ( data.get() == HTTPMessage::charCR ) {
180  state = psHTTPVersionLF;
181  parseResult.setSystemError( data.next() );
182  if ( !parseResult.ok() ) return parseResult;
183  } else {
184  return { peExpectCRLF, common::SystemError::ecOK };
185  }
186  break;
187 
188  case psHTTPVersionLF:
189  if ( data.get() == HTTPMessage::charLF ) {
190  state = psDone;
191  parseResult.setSystemError( data.next() );
192  if ( !parseResult.ok() ) return parseResult;
193  } else {
194  return { peExpectCRLF, common::SystemError::ecOK };
195  }
196  break;
197 
198  default: throw_SystemException( common::Puts() <<
199  "HTTPRequest::HTTPRequestLine::parse unhandled state " <<
200  state, 0 );
201 
202  }
203  }
204  if ( state == psDone ) {
205  method_ = methodFromString( method_string );
206  if ( method_ == meINVALID ) return { peInvalidMethod, common::SystemError::ecOK };
207  } else return { peInvalidRequestLine, common::SystemError::ecOK };
208  return parseResult;
209  }
