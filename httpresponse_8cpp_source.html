32  HTTPMessage::ParseResult HTTPResponse::parse( VirtualReadBuffer &data ) {
33  ParseResult parseResult = response_line_.parse( data );
34  if ( parseResult.ok() ) {
35  if ( data.get() == charCR ) {
36  parseResult.setSystemError( data.next() );
37  if ( parseResult.ok() && data.get() == charLF ) {
38  parseResult.setSystemError( data.next() );
39  if ( parseResult.ok() ) return parseBody( data ); else return parseResult;
40  } else return { peOk, common::SystemError::ecOK };
41  } else {
42  // these are headers
43  parseResult = parseHeaders( data );
44  if ( !parseResult.ok() ) return parseResult;
45  if ( data.get() == charLF ) {
46  parseResult.setSystemError( data.next() );
47  if ( parseResult.ok() ) {
48  return parseBody( data );
49  } else if ( parseResult.eof() ) return { peOk, common::SystemError::ecOK };
50  else return parseResult;
51  } else return { peOk, common::SystemError::ecOK };
52  }
53  }
54  return parseResult;
55  }

82  HTTPFragment::ParseResult HTTPResponse::HTTPResponseLine::parse( VirtualReadBuffer& data ) {
83  ParseResult parseResult;
84  parseResult = version_.parse( data );
85  if ( parseResult.ok()) {
86  parseResult.setSystemError( eatSpace( data ) );
87  if ( !parseResult.ok() ) return parseResult;
88  unsigned int i;
89  parseResult = parseUInt( data, i );
90  if ( parseResult.ok() ) {
91  http_code_ = static_cast<HTTPCode>(i);
92  // ignore the error string (we already have the HTTP code) but we do expect CRLF.
93  while ( parseResult.ok() && data.get() != charCR ) {
94  parseResult.setSystemError( data.next() );
95  if ( ! parseResult.ok() ) return parseResult;
96  }
97  if ( data.get() != charCR )
98  return { peExpectCRLF, common::SystemError::ecOK };
99  else {
100  parseResult.setSystemError( data.next() );
101  if ( ! parseResult.ok() ) return parseResult;
102  }
103  if ( data.get() != charLF )
104  return { peExpectCRLF, common::SystemError::ecOK };
105  else {
106  parseResult.setSystemError( data.next() );
107  if ( parseResult.ok() ) return parseResult;
108  }
109  }
110  }
111  return parseResult;
112  }
