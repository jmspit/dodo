51  SystemError KVStore::open( const std::filesystem::path &path, ShareMode mode ) {
52  fd_ = ::open( path.c_str(), O_RDWR | O_DSYNC, 0600 );
53  if ( fd_ == -1 ) throw_SystemException( "open of file '" << path << "' failed", errno );
54  ssize_t size = getFileSize( path );
55  switch ( mode ) {
56  case ShareMode::Private:
57  address_ = mmap( nullptr, size, PROT_READ | PROT_WRITE, MAP_PRIVATE, fd_, 0 );
58  waitReadLock_ = &KVStore::waitReadLockPrivate;
59  waitWriteLock_ = &KVStore::waitWriteLockPrivate;
60  break;
61  case ShareMode::Shared:
62  address_ = mmap( nullptr, size, PROT_READ | PROT_WRITE, MAP_SHARED, fd_, 0 );
63  waitReadLock_ = &KVStore::waitReadLockShared;
64  waitWriteLock_ = &KVStore::waitWriteLockShared;
65  break;
66  case ShareMode::Clustered:
67  address_ = mmap( nullptr, size, PROT_READ | PROT_WRITE, MAP_SHARED, fd_, 0 );
68  waitReadLock_ = &KVStore::waitReadLockShared;
69  waitWriteLock_ = &KVStore::waitWriteLockShared;
70  break;
71  }
72  if ( address_ == MAP_FAILED ) throw_SystemException( "mmap failed", errno );
73  return SystemError::ecOK;
74  }

76  SystemError KVStore::init( const std::filesystem::path &path, size_t blocksize, size_t blocks, ShareMode mode ) {
77  fd_ = ::open( path.c_str(), O_RDWR | O_DSYNC | O_CREAT | O_TRUNC, 0600 );
78  if ( fd_ == -1 ) throw_SystemException( "creation of file '" << path << "' failed", errno );
79  // size the file
80  blocksize_ = ( blocksize / sysconf(_SC_PAGESIZE) ) * sysconf(_SC_PAGESIZE);
81  size_t l_blocks = blocks;
82  if ( l_blocks < min_blocks_ ) l_blocks = min_blocks_;
83  size_t size = blocksize_ * l_blocks;
84  lseek( fd_, size - 1, SEEK_SET );
85  write( fd_, "A", 1 );
86  switch ( mode ) {
87  case ShareMode::Private:
88  address_ = mmap( nullptr, size, PROT_READ | PROT_WRITE, MAP_PRIVATE, fd_, 0 );
89  break;
90  case ShareMode::Shared:
91  case ShareMode::Clustered:
92  address_ = mmap( nullptr, size, PROT_READ | PROT_WRITE, MAP_SHARED, fd_, 0 );
93  break;
94  }
95  if ( address_ == MAP_FAILED ) throw_SystemException( "mmap failed", errno );
96 
97  // initialize header
98  FileHeader fileheader( address_ );
99  fileheader.getBlock().block_header.zero(blocksize_);
100  fileheader.getBlock().block_header.blockid = 0;
101  fileheader.getBlock().block_header.blocktype = BlockType::btFileHeader;
102  fileheader.getBlock().blocksize = blocksize_;
103  fileheader.getBlock().blocks = l_blocks;
104  fileheader.getBlock().magic = magic;
105  fileheader.getBlock().version = version;
106  fileheader.getBlock().created = time(nullptr);
107  fileheader.getBlock().name_sz = 0;
108  fileheader.getBlock().description_sz = 0;
109  fileheader.getBlock().contact_sz = 0;
110 
111  fileheader.setInfo( "Dodo test name", "Dodo test description", "Dodo test contact" );
112 
113  fileheader.getBlock().block_header.crc32 = fileheader.getBlock().block_header.calcCRC32( blocksize_ );
114 
115  // initialize TOC
116  TOCHeader* toc_header = reinterpret_cast<TOCHeader*>( getBlockAddress( 1 ) );
117  toc_header->block_header.blockid = 1L;
118  toc_header->block_header.blocktype = btTOC;
119  toc_header->next_toc = 0L;
120 
121  // initialize blocks
122  for ( BlockId b = 2; b < fileheader.getBlock().blocks; b++ ) {
123  BlockHeader* bh = getBlockAddress( b );
124  bh->zero(blocksize_);
125  bh->blockid = b;
126  bh->blocktype = btFree;
127  }
128 
129  SystemError rc = msync( address_, size, MS_SYNC );
130 
131  if ( rc != 0 ) throw_SystemException( "msync failed", errno );
132  return SystemError::ecOK;
133  }

170  bool KVStore::analyze( std::ostream &os ) {
171  bool ok = true;
172  os << "FileHeader" << endl;

175  Tester tester( os );

178 
179  ok = tester.test( common::Puts() << "magic " << fileheader.getBlock().magic,
180  [&fileheader,this](){ return fileheader.getBlock().magic == magic; } );
181 
182  ok = tester.test( common::Puts() << "blocksize " << fileheader.getBlock().blocksize,
183  [&fileheader](){ return fileheader.getBlock().blocksize % sysconf(_SC_PAGESIZE) == 0; } );
184 
185  ok = tester.test( common::Puts() << "file version " << fileheader.getBlock().version,
186  [&fileheader,this](){ return fileheader.getBlock().version = version; } );
187 
188  ok = tester.test( common::Puts() << "crc32 0x" << common::Puts::hex() << fileheader.getBlock().block_header.calcCRC32( blocksize_ ) << common::Puts::dec(),
189  [&fileheader,this](){ return fileheader.getBlock().block_header.calcCRC32( blocksize_ ) == fileheader.getBlock().block_header.crc32 ; } );
190 
191  struct stat st;
192  fstat( fd_, &st );
193  ok = tester.test( common::Puts() << "file size " << st.st_size,
194  [&fileheader,st,this](){ return fileheader.getBlock().blocksize * fileheader.getBlock().blocks == static_cast<uint64_t>( st.st_size ); } );
195 
196  os << "created " << fileheader.getBlock().created << std::endl;
197  os << "blocks " << fileheader.getBlock().blocks << std::endl;
198  std::string name, description, contact;
199  fileheader.getInfo( name, description, contact );
200  os << "name " << name << std::endl;
201  os << "description " << description << std::endl;
202  os << "contact " << contact << std::endl;
203 
204  return ok;
