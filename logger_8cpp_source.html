46  Logger::Logger( const Config& config ) {
47  rotate_throttle_counter_ = 0;
48  destinations_ = Console;
49  char hostname[256];
50  gethostname( hostname, 256 );
51  hostname_ = hostname;
52  threads::Mutexer lock( mutex_ );
53  YAML::Node &yaml = config.getConfig()->getYAML();
54  levels_[Console] = StringAsLogLevel( yaml["dodo"]["common"]["logger"]["console"]["level"].as<std::string>() );
55 
56  if ( yaml["dodo"]["common"]["logger"]["file"] ) {
57  destinations_ = destinations_ | File;
58  getFileParams(config);
59  }
60 
61  if ( yaml["dodo"]["common"]["logger"]["syslog"] ) {
62  destinations_ = destinations_ | Syslog;
63  if ( yaml["dodo"]["common"]["logger"]["syslog"]["facility"] ) {
64  syslog_params_.facility = yaml["dodo"]["common"]["logger"]["syslog"]["facility"].as<int>();
65  } else syslog_params_.facility = 1;
66 
67  levels_[Syslog] = StringAsLogLevel( yaml["dodo"]["common"]["logger"]["syslog"]["level"].as<std::string>() );
68  if ( levels_[Syslog] > LogLevel::Info ) throw_Exception( "syslog LogLevel cannot be higher than info" );
69  }
70 
71  }

90  void Logger::log( LogLevel level, const std::string &message ) {
91  threads::Mutexer lock( mutex_ );
92  std::string entry = "";
93 
94  if ( level <= levels_[Console] ) {
95  entry = formatMessage( level, message );
96  std::cout << entry << std::endl;
97  }
98 
99  if ( destinations_ | File && level <= levels_[File] ) {
100  checkRotate();
101  if ( entry.length() == 0 ) entry = formatMessage( level, message );
102  file_params_.file << entry << std::endl;
103  file_params_.filesize += entry.length() + 1;
104  }
105 
106  if ( destinations_ | Syslog && level <= levels_[Syslog] && level <= LogLevel::Info ) {
107  std::stringstream ss;
108  ss << Config::getConfig()->getAppName() << "[" << syscall(SYS_gettid) << "]: ";
109  ss << LogLevelAsString(level, true) << ": " << message;
110  syslog( syslog_params_.facility | mapLeveltoSyslog(level), "%s", ss.str().c_str() );
111  }
112  }

155  void Logger::checkRotate() {
156  if ( file_params_.filesize > file_params_.max_size_mib * 1024 * 1024 ) {
157  file_params_.file.close();
158  std::stringstream ss;
159  struct timeval tv;
160  gettimeofday( &tv, nullptr );
161  ss << file_params_.active_log << "." << formatDateTimeUTC( tv );
162  rename( file_params_.active_log.c_str(), ss.str().c_str() );
163  file_params_.file.open( file_params_.active_log, std::ofstream::out | std::ofstream::app );
164  file_params_.filesize = getFileSize( file_params_.active_log );
165 
166  std::set<std::string> trail;
167  std::regex log_regex( "^" + file_params_.active_log + "\\..*" );
168  for ( const auto & entry : std::filesystem::directory_iterator( file_params_.directory ) ) {
169  if ( std::regex_match( entry.path().string(), log_regex ) ) {
170  trail.insert( entry.path().string() );
171  }
172  }
173 
174  while ( trail.size() > file_params_.max_file_trail ) {
175  std::string path = *trail.begin();
176  unlink( path.c_str() );
177  trail.erase( trail.begin() );
178  }
179  }
180  }

182  void Logger::getFileParams( const Config& config ) {
183  YAML::Node &yaml = config.getConfig()->getYAML();
184  levels_[File] = StringAsLogLevel( yaml["dodo"]["common"]["logger"]["file"]["level"].as<std::string>() );
185  file_params_.directory = yaml["dodo"]["common"]["logger"]["file"]["directory"].as<std::string>();
186 
187  if ( yaml["dodo"]["common"]["logger"]["file"]["max-size-mib"] )
188  file_params_.max_size_mib = yaml["dodo"]["common"]["logger"]["file"]["max-size-mib"].as<size_t>();
189  else
190  file_params_.max_size_mib = 10;
191 
192  if ( yaml["dodo"]["common"]["logger"]["file"]["max-file-trail"] )
193  file_params_.max_file_trail = yaml["dodo"]["common"]["logger"]["file"]["max-file-trail"].as<size_t>();
194  else
195  file_params_.max_file_trail = 4;
196 
197  if ( !directoryExists( file_params_.directory ) )
198  throw_Exception( "directory '" << file_params_.directory << "' does not exist" );
199  if ( !directoryWritable( file_params_.directory ) )
200  throw_Exception( "directory '" << file_params_.directory << "' does not allow write access" );
201  file_params_.active_log = file_params_.directory + "/" + config.getAppName() + ".log";
202  file_params_.file.open( file_params_.active_log, std::ofstream::out | std::ofstream::app );
203  file_params_.filesize = getFileSize( file_params_.active_log );
204  if ( !file_params_.file.good() )
205  throw_Exception( "failed to open '" << file_params_.active_log << "' for writing" );
206  }
