33  void OctetArray::reserve( size_t sz ) {
34  size_t chunkedsz = ( sz / alloc_block + 1 ) * alloc_block;
35  if ( sz == size ) return;
36  if ( array ) {
37  if ( sz > allocated_size ) {
38  array = static_cast< Octet*>( std::realloc( array, chunkedsz ) );
39  if ( array || sz == 0 ) {
40  size = sz;
41  allocated_size = chunkedsz;
42  if ( !sz ) array = nullptr;
43  } else throw_SystemException( "realloc of " << sz << " bytes failed", errno );
44  } else {
45  size = sz;
46  }
47  } else {
48  array = static_cast< Octet*>( std::malloc( chunkedsz ) );
49  if ( array ) {
50  size = sz;
51  allocated_size = chunkedsz;
52  } else throw_SystemException( "malloc of " << sz << " bytes failed", errno );
53  }
54  }

114  OctetArray& OctetArray::decodeBase64( const std::string& src ) {
115  Octet* temp = (Octet*)std::malloc( src.length() ); // too much, but safe, and temp anyway
116 
117 
118  EVP_ENCODE_CTX* ctx = EVP_ENCODE_CTX_new();
119  if ( !ctx ) throw_Exception( "EVP_ENCODE_CTX_new failed" );
120  EVP_DecodeInit( ctx );
121  int len = 0;
122  int rc = EVP_DecodeUpdate( ctx, temp, &len, (const unsigned char*)src.c_str(), (int)src.length() );
123  if ( rc == -1 ) throw_Exception( "EVP_DecodeUpdate failed" );
124  int actsize = len;
125  rc = EVP_DecodeFinal( ctx, temp + len, &len );
126  if ( rc != 1 ) throw_Exception( "EVP_DecodeFinal failed" );
127  actsize += len;
128  EVP_ENCODE_CTX_free( ctx );
129 
130  this->reserve( actsize );
131  memcpy( array, temp, actsize );
132  std::free( temp );
133  return *this;
134  }

162  OctetArray::MatchType OctetArray::match( const OctetArray& other, size_t index, size_t &octets ) {
163  size_t local_size = size - index;
164  size_t match_size = std::min( local_size, other.size );
165  int cmp = memcmp( array, other.array, match_size );
166  if ( cmp == 0 ) {
167  if ( local_size < other.size ) {
168  octets = local_size;
169  return MatchType::Partial;
170  } else {
171  octets = other.size;
172  return MatchType::Full;
173  }
174  } else {
175  octets = 0;
176  return MatchType::Mismatch;
177  }
178  }
