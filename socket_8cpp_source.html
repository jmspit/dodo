36  common::SystemError Socket::send( const void* buf, ssize_t len, bool more ) {

38  if ( more ) flags = flags | MSG_MORE;
39  ssize_t left = len;
40  ssize_t rc = 0;
41  const char* cp = (const char*)buf;
42  while ( left > 0 ) {
43  rc = ::send( socket_, cp, left, flags );
44  if ( rc < 0 ) break;
45  cp += rc;
46  left -= rc;
47  }
48  if ( rc < 0 ) {
