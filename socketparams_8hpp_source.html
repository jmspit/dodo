195  /**
196  * Return the AddressFamily name as a string.
197  * @param family The AddressFamily to convert.
198  * @return The string.
199  */
200  static string familyString( AddressFamily family ) {
201  stringstream ss;
202  switch ( family ) {
203  case AF_LOCAL : ss << "AF_LOCAL"; break;
204  case AF_INET : ss << "AF_INET"; break;
205  case AF_INET6 : ss << "AF_INET6"; break;
206  case AF_IPX : ss << "AF_IPX"; break;
207  case AF_NETLINK : ss << "AF_NETLINK"; break;
208  case AF_X25 : ss << "AF_X25"; break;
209  case AF_AX25 : ss << "AF_AX25"; break;
210  case AF_ATMPVC : ss << "AF_ATMPVC"; break;
211  case AF_APPLETALK : ss << "AF_APPLETALK"; break;
212  case AF_PACKET : ss << "AF_PACKET"; break;
213  case AF_ALG : ss << "AF_ALG"; break;
214  case AF_UNSPEC : ss << "AF_UNSPEC"; break;
215  default : ss << "unhandled protocol family"; break;
216  }
217  return ss.str();
218  };
