43  static void sqlite_ext_pow(sqlite3_context *context, int argc, sqlite3_value **argv) {
44  double r1 = 0.0;
45  double r2 = 0.0;
46  double val;
47 
48  assert( argc==2 );
49 
50  if( sqlite3_value_type(argv[0]) == SQLITE_NULL || sqlite3_value_type(argv[1]) == SQLITE_NULL ){
51  sqlite3_result_null(context);
52  } else {
53  r1 = sqlite3_value_double(argv[0]);
54  r2 = sqlite3_value_double(argv[1]);
55  errno = 0;
56  val = pow(r1,r2);
57  if (errno == 0) {
58  sqlite3_result_double(context, val);
59  } else {
60  sqlite3_result_error(context, strerror(errno), errno);
61  }
62  }
63  }

125  Database::Database( const std::string &filename, WaitHandler handler ) {
126  database_ = NULL;
127  int r = sqlite3_open_v2( filename.c_str(), &database_, SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE | SQLITE_OPEN_NOMUTEX, NULL );
128  if ( r != SQLITE_OK ) {
129  throw_Exception( sqlite3_errmsg( database_ ) );
130  }
131  int eTextRep = SQLITE_UTF8;
132  #if SQLITE_VERSION_NUMBER > 3008003
133  eTextRep |= SQLITE_DETERMINISTIC;
134  #endif
135  r = sqlite3_create_function_v2( database_,
136  "power",
137  2,
138  eTextRep,
139  NULL,
140  sqlite_ext_pow,
141  NULL,
142  NULL,
143  NULL );
144  if ( r != SQLITE_OK ) {
145  throw_Exception( sqlite3_errmsg( database_ ) );
146  }
147  r = sqlite3_create_function_v2( database_,
148  "ceil",
149  1,
150  eTextRep,
151  NULL,
152  sqlite_ext_ceil,
153  NULL,
154  NULL,
155  NULL );
156  if ( r != SQLITE_OK ) {
157  throw_Exception( sqlite3_errmsg( database_ ) );
158  }
159  r = sqlite3_create_function_v2( database_,
160  "floor",
161  1,
162  eTextRep,
163  NULL,
164  sqlite_ext_floor,
165  NULL,
166  NULL,
167  NULL );
168  if ( r != SQLITE_OK ) {
169  throw_Exception( sqlite3_errmsg( database_ ) );
170  }
171  r = sqlite3_create_function_v2( database_,
172  "log2",
173  1,
174  eTextRep,
175  NULL,
176  sqlite_ext_log2,
177  NULL,
178  NULL,
179  NULL );
180  if ( r != SQLITE_OK ) {
181  throw_Exception( sqlite3_errmsg( database_ ) );
182  }
183 
184 
185  if ( handler ) sqlite3_busy_handler( database_, handler, database_ );
186  }
