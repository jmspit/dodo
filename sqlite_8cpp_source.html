38  static void sqlite_ext_pow(sqlite3_context *context, int argc, sqlite3_value **argv) {
39  assert( argc==2 );
40  if( sqlite3_value_type(argv[0]) == SQLITE_NULL || sqlite3_value_type(argv[1]) == SQLITE_NULL ) {
41  sqlite3_result_null(context);
42  } else {
43  errno = 0;
44  double r1 = sqlite3_value_double(argv[0]);
45  double r2 = sqlite3_value_double(argv[1]);
46  double val = pow(r1,r2);
47  if (errno == 0) {
48  sqlite3_result_double( context, val );
49  } else {
50  sqlite3_result_error( context, strerror(errno), errno );
51  }
52  }
53  }

109  int r = sqlite3_open_v2( filename.c_str(), &database_, SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE | SQLITE_OPEN_NOMUTEX, nullptr );
