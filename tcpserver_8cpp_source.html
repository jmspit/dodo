46  void TCPServer::run() {
47  has_stopped_ = false;
48  network::TCPListener::SocketWork* sockmap = 0;
49  struct timeval now, last_snap;
50  gettimeofday( &last_active_, NULL );
51  gettimeofday( &now, NULL );
52  snapRUsage();
53  gettimeofday( &last_snap, NULL );
54  state_ = ssWait;
55  try {
56  do {
57  if ( listener_.waitForActivity( this ) ) {
58 
59  busy_ = true;
60  gettimeofday( &last_active_, NULL );
61 
62  state_ = ssAwoken;
63 
64  do {
65 
66  sockmap = listener_.popWork();
67 
68  if ( sockmap ) {
69 
70  log_Debug( "TCPServer::run notify wakeup " <<
71  sockmap->socket->debugString() << " state " << sockmap->state );
72 
73  TCPListener::SockState completion_state = TCPListener::SockState::None;
74 
75  if ( sockmap->state & TCPListener::SockState::New ) {
76  log_Debug( "TCPServer::run ssNew " <<
77  sockmap->socket->debugString() << " state " << sockmap->state );
78  bool ok = false;
79  state_ = ssHandshake;
80  try {
81  ssize_t received = 0;
82  ssize_t sent = 0;
83  ok = handShake( sockmap->socket, received, sent );
84  listener_.addReceivedSentBytes( received, sent );
85  state_ = ssHandshakeDone;
86  completion_state |= TCPListener::SockState::New;
87  if ( !ok ) {
88  completion_state |= TCPListener::SockState::Shut;
89  log_Error( "TCPServer::run handshake failure socket " <<
90  sockmap->socket->debugString() );
91  }
92  }
93  catch ( std::exception &e ) {
94  log_Error( "TCPServer::run exception in handshake " << e.what()
95  << " socket " << sockmap->socket->debugString() );
96  }
97  catch ( ... ) {
98  log_Error( "TCPServer::run unhandled exception in handshake " <<
99  sockmap->socket->debugString() );
100  }
101  }
102 
103  if ( sockmap->state & TCPListener::SockState::Read ) {
104  log_Debug( "TCPServer::run ssRead " <<
105  sockmap->socket->debugString() << " state " << sockmap->state );
106  bool ok = false;
107  state_ = ssReadSocket;
108  try {
109  ssize_t received = 0;
110  ssize_t sent = 0;
111 
112  SystemError error = sockmap->data->readBuffer( sockmap->socket, received );
113  ok = (error == SystemError::ecOK);
114  error = readSocket( *sockmap, sent );
115  ok = ok && ( error == SystemError::ecOK || error == SystemError::ecEAGAIN );
116  listener_.addReceivedSentBytes( received, sent );
117  state_ = ssReadSocketDone;
118  completion_state |= TCPListener::SockState::Read;
119  if ( !ok ) {
120  completion_state |= TCPListener::SockState::Shut;
121  if ( error != SystemError::ecECONNABORTED ) {
122  log_Error( "TCPServer::run readSocket failure socket " <<
123  sockmap->socket->getFD() << " client " <<
124  sockmap->socket->getPeerAddress().asString(true) );
125  }
126  }
127  }
128  catch ( std::exception &e ) {
129  log_Error( "TCPServer::run exception in ssReadSocket " << e.what()
130  << " socket " << sockmap->socket->getFD() );
131  }
132  catch ( ... ) {
133  log_Error( "TCPServer::run unhandled exception in ssReadSocket socket " <<
134  sockmap->socket->getFD() );
135  }
136  state_ = ssReadSocketDone;
137  }
138 
139  if ( sockmap->state & TCPListener::SockState::Shut ) {
140  log_Debug( "TCPServer::run ssShut " <<
141  sockmap->socket->debugString() << " state " << sockmap->state );
142  state_ = ssShutdown;
143  try {
144  shutDown( sockmap->socket );
145  state_ = ssShutdownDone;
146  completion_state |= TCPListener::SockState::Shut;
147  }
148  catch ( std::exception &e ) {
149  log_Error( "TCPServer::run exception in ssShutdown " << e.what()
150  << " socket " << sockmap->socket->debugString() );
151  }
152  catch ( ... ) {
153  log_Error( "TCPServer::run unhandled exception in ssShutdown " <<
154  sockmap->socket->debugString() );
155  }
156  }
157  state_ = ssReleaseWork;
158  listener_.releaseWork( { sockmap->socket, completion_state } );
159  state_ = ssReleaseWorkDone;
160  }
161 
162  } while ( sockmap );
163 
164  busy_ = false;
165  state_ = ssWait;
166  }
167  gettimeofday( &now, NULL );
168  if ( common::getSecondDiff( last_snap, now ) > 0.2 ) {
169  snapRUsage();
170  gettimeofday( &last_snap, NULL );
171  }
172 
173  } while ( !request_stop_ );
174  log_Debug( "TCPServer::run stopping" );
175  }
176  catch ( const std::exception &e ) {
177  log_Fatal( "TCPServer::run caught unhandled exception " << e.what() );
178  }
179  catch ( ... ) {
180  log_Fatal( "TCPServer::run caught std::exception " );
181  }
182  has_stopped_ = true;
183  }
