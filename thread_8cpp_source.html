33  void* Thread::thread_method( void* context ) {
34  Thread* t = ((Thread*)context);
35  t->tid_ = static_cast<pid_t>(syscall(SYS_gettid));
36  gettimeofday( &t->start_time_, NULL );
37  t->snapRUsage();
38  if ( t ) t->run();
39  return 0;
40  }
