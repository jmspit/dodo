64  void TLSContext::construct( const PeerVerification& peerverficiation,
65  const TLSVersion& tlsversion,
66  bool enableSNI,
67  bool allowSANWildcards ) {
68  tlsversion_ = tlsversion;
69  peerverficiation_ = peerverficiation;
70  enable_sni_ = enableSNI;
71  allow_san_wildcards_ = allowSANWildcards;
72  passphrase_ = "";
73  tlsctx_ = nullptr;
74  long rc = 0;
75  tlsctx_ = SSL_CTX_new( TLS_method() );
76  if ( !tlsctx_ ) throw_ExceptionObject( "SSL_CTX_new failed"
77  << common::Puts::endl() << common::getSSLErrors( '\n' ), this );
78  switch( tlsversion_ ) {
79  case TLSVersion::tls1_1 :
80  rc = SSL_CTX_set_min_proto_version( tlsctx_, TLS1_1_VERSION );
81  break;
82  case TLSVersion::tls1_2 :
83  rc = SSL_CTX_set_min_proto_version( tlsctx_, TLS1_2_VERSION );
84  break;
85  case TLSVersion::tls1_3 :
86  rc = SSL_CTX_set_min_proto_version( tlsctx_, TLS1_3_VERSION );
87  break;
88  default:
89  rc = SSL_CTX_set_min_proto_version( tlsctx_, TLS1_1_VERSION );
90  break;
91  }
92  if ( rc == 0 ) throw_ExceptionObject( "SSL_CTX_set_min_proto_version failed"
93  << common::Puts::endl() << common::getSSLErrors( '\n' ), this );
94 
95  SSL_CTX_set_default_passwd_cb( tlsctx_, pem_passwd_cb );
96  SSL_CTX_set_default_passwd_cb_userdata( tlsctx_, this );
97 
98  if ( peerverficiation == PeerVerification::pvVerifyNone ) {
99  SSL_CTX_set_verify( tlsctx_, SSL_VERIFY_NONE, nullptr );
100  } else {
101  SSL_CTX_set_verify( tlsctx_, SSL_VERIFY_PEER | SSL_VERIFY_FAIL_IF_NO_PEER_CERT, nullptr );
102  }
103 
104  rc = SSL_CTX_set_default_verify_paths(tlsctx_);
105  if ( rc == 0 ) throw_ExceptionObject( "SSL_CTX_set_default_verify_paths failed"
106  << common::Puts::endl() << common::getSSLErrors( '\n' ), this );
107 
108  }

115  void TLSContext::loadPEMIdentity( const std::string& certfile,
116  const std::string& keyfile,
117  const std::string& passphrase ) {
118  passphrase_ = passphrase;
119  if ( SSL_CTX_use_certificate_file( tlsctx_, certfile.c_str(), SSL_FILETYPE_PEM ) != 1 ) {
120  throw_ExceptionObject( common::getSSLErrors( '\n' ), this );
121  }
122  if ( SSL_CTX_use_PrivateKey_file( tlsctx_, keyfile.c_str(), SSL_FILETYPE_PEM ) != 1 ) {
123  throw_ExceptionObject( common::getSSLErrors( '\n' ), this );
124  }
125  if ( !SSL_CTX_check_private_key( tlsctx_ ) ) {
126  throw_ExceptionObject( common::getSSLErrors( '\n' ), this );
127  }
128  passphrase_ = "";
129  }

131  void TLSContext::loadPKCS12( const std::string &p12file,
132  const std::string &p12passphrase ) {
133  passphrase_ = "";
134  PKCS12 *p12 = nullptr;
135  FILE *fp = 0;
136  if ( ( fp = fopen( p12file.c_str(), "rb" ) ) )
137  {
138  p12 = d2i_PKCS12_fp( fp, NULL );
139  if ( p12 ) {
140  EVP_PKEY *pkey = nullptr;
141  X509 *cert = nullptr;
142  STACK_OF(X509) *ca = nullptr;
143  if ( PKCS12_parse( p12, p12passphrase.c_str(), &pkey, &cert, &ca) ) {
144 
145  try {
146 
147  if ( SSL_CTX_use_certificate( tlsctx_, cert ) != 1 )
148  throw_ExceptionObject( "cannot use certificate from '" << p12file << "'"
149  << common::Puts::endl() << common::getSSLErrors( '\n' ), this );
150 
151  if ( SSL_CTX_use_PrivateKey( tlsctx_, pkey ) != 1 || !pkey )
152  throw_ExceptionObject( "cannot use private key from '" << p12file << "'"
153  << common::Puts::endl() << common::getSSLErrors( '\n' ), this );
154 
155  if ( !SSL_CTX_check_private_key( tlsctx_ ) )
156  throw_ExceptionObject( "invalid private key in '" << p12file << "'"
157  << common::Puts::endl() << common::getSSLErrors( '\n' ), this );
158 
159  if ( !SSL_CTX_set0_chain( tlsctx_, ca ) ) {
160 
161  throw_ExceptionObject( "cannot use certificate chain from '" << p12file << "'"
162  << common::Puts::endl() << common::getSSLErrors( '\n' ), this );
163  }
164  if ( cert ) X509_free( cert );
165  if ( pkey ) EVP_PKEY_free( pkey );
166  if ( p12 ) PKCS12_free( p12 );
167  fclose( fp );
168  }
169  catch ( ... ) {
170  if ( cert ) X509_free( cert );
171  if ( pkey ) EVP_PKEY_free( pkey );
172  if ( p12 ) PKCS12_free( p12 );
173  fclose( fp );
174  throw;
175  }
176  } else throw_ExceptionObject( "cannot parse PKCS12 file '" << p12file << "'"
177  << common::Puts::endl() << common::getSSLErrors( '\n' ), this );
178  } else throw_ExceptionObject( "cannot read PKCS12 file '" << p12file << "'"
179  << common::Puts::endl() << common::getSSLErrors( '\n' ), this );
180  } else throw_ExceptionObject( "cannot open'" << p12file << "'"
181  << common::Puts::endl() << common::getSSLErrors( '\n' ), this );
182  }

192  TLSContext* tlsctx = static_cast<TLSContext*>(userdata);
