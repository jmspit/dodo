63  /**
64  * The TLS peer verification method.
65  *
66  * | PeerVerification | SSL_CTX_set_verify |
67  * | -----------------| ----------------------------------------------------|
68  * | pvVerifyNone | SSL_VERIFY_NONE |
69  * | pvVerifyPeer | SSL_VERIFY_PEER \| SSL_VERIFY_FAIL_IF_NO_PEER_CERT |
70  * | pvVerifyFQDN | SSL_VERIFY_PEER \| SSL_VERIFY_FAIL_IF_NO_PEER_CERT |
71  * | pvVerifyCustom | SSL_VERIFY_PEER \| SSL_VERIFY_FAIL_IF_NO_PEER_CERT |
72  *
73  * | PeerVerification | Server / accept | Client / connect |
74  * |------------------|-----------------|------------------|
75  * | pvVerifyNone | encryption of traffic | encryption of traffic |
76  * | pvVerifyPeer | pvVerifyNone + client must present trusted cert | pvVerifyNone + server must present trusted cert |
77  * | pvVerifyFQDN | pvVerifyPeer | pvVerifyPeer + X509Certificate::verifyName() |
78  * | pvVerifyCustom | pvVerifyPeer + custom | pvVerifyPeer + custom |
79  *
80  * @see https://www.openssl.org/docs/man1.1.1/man3/SSL_CTX_set_verify.html
81  */
82  enum class PeerVerification {
83  pvVerifyNone, /**< No peer verification - transmission is encrypted, peer cis accepted even if
84  peer certificate is invalid and can read all data sent.*/
85  pvVerifyPeer, /**< The peer must have a trusted certificate (unless a anonymous cipher is used). */
86  pvVerifyFQDN, /**< As pvVerifyPeer, but the remote DNS name must match either the peer cert commonname
87  or match one of the peer cert subjectAltNames */
88  pvVerifyCustom, /**< As pvVerifyPeer, but with custom certificate validation logic.*/
89  };

91  /**
92  * Construct a TLS context.
93  * @param peerverficiation The PeerVerification method to use.
94  * @param tlsversion The TLS version to use. Use of default is less future code hassle.
95  * @param enableSNI Enable the Server Name Indication extension. Note that this exposes the target hostname
96  * of TLSSocket connections as the hostname is sent unencrypted, facilitated all kinds of evil such as
97  * censorship. Use only when you must connect to a server that requires it.
98  *
99  * @see https://en.wikipedia.org/wiki/Server_Name_Indication
100  */
