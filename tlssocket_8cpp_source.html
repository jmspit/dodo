56  common::SystemError TLSSocket::connect( const Address &address ) {
57  common::SystemError error = BaseSocket::connect( address );
58  if ( error == common::SystemError::ecOK ) {
59  auto rc = SSL_connect( ssl_ );
60  if ( rc != 1 ) {
61  auto ssl_error_code = SSL_get_error( ssl_, rc );
62  switch ( ssl_error_code ) {
63  case SSL_ERROR_NONE : return common::SystemError::ecSSL_ERROR_NONE;
64  case SSL_ERROR_ZERO_RETURN : return common::SystemError::ecSSL_ERROR_ZERO_RETURN;
65  case SSL_ERROR_WANT_READ : return common::SystemError::ecSSL_ERROR_WANT_READ;
66  case SSL_ERROR_WANT_WRITE : return common::SystemError::ecSSL_ERROR_WANT_WRITE;
67  case SSL_ERROR_WANT_CONNECT : return common::SystemError::ecSSL_ERROR_WANT_CONNECT;
68  case SSL_ERROR_WANT_ACCEPT : return common::SystemError::ecSSL_ERROR_WANT_ACCEPT;
69  case SSL_ERROR_WANT_X509_LOOKUP : return common::SystemError::ecSSL_ERROR_WANT_X509_LOOKUP;
70  case SSL_ERROR_SSL :
71  default: throw_Exception( ssl_error_code << " " << common::getSSLErrors( '\n' ) );
72  }
73  }
74  if ( tlscontext_.getPeerVerification() == TLSContext::PeerVerification::pvVerifyFQDN ) {
75  if ( X509Certificate::verifySAN( getPeerCertificate(), { network::X509Common::SANType::stDNS, peer_name_.san_name } ) ) {
76  return common::SystemError::ecOK;
77  } else return common::SystemError::ecSSL_ERROR_PEERVERIFICATION;
78  } else return common::SystemError::ecOK;
79  } else return error;
80  }

82  TLSSocket* TLSSocket::accept() {
83  auto rc = SSL_accept( ssl_ );
84  if ( rc <= 0 ) {

86  log_Error( ssl_error_code << " " << common::getSSLErrors( '\n' ) );
87  return nullptr;
88  } else {
89  TLSSocket* ret = new TLSSocket( rc, tlscontext_, { X509Common::SANType::stDNS, "" } );
90  return ret;
91  }
92  return nullptr;
